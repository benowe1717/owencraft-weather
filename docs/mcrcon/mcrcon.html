 75    def _ansi_decode(self, output: str) -> str:
 76        """
 77        Strip all ANSI characters from the output
 78
 79        All MCRCON packets have ANSI characters in them. This method
 80        strips out all of those characters by doing a find and replace.
 81
 82        Args:
 83            output: The direct output from the self._run_command() method
 84
 85        Returns:
 86            A string with all ANSI characters stripped from the output
 87
 88        Raises:
 89            None
 90        """
 91        pattern = r'(?:\x1B[@-_]|[\x80-\x9F])[0-?]*[ -/]*[@-~]'
 92        return re.sub(pattern, '', output)

 94    def _mcrcon_strip(self, output: str) -> str:
 95        """
 96        Remove lines of data when using MCRCON from multiple sources
 97
 98        When using MCRCON from multiple sources, the output from MCRCON
 99        gets combined all into a single stream. So this strips out
100        all the possible lines that aren't relevant.
101
102        Args:
103            output: The direct output from the self._run_command() method
104
105        Returns:
106            A string with all configured lines and characters stripped from
107            the output
108
109        Raises:
110            None
111        """
112        lines: list[str] = [
113            'Automatic saving is now disabled',
114            'Automatic saving is now enabled',
115            'Saved the game'
116        ]
117        for line in lines:
118            output = output.replace(line, '')
119
120        return output

122    def _run_command(self, cmd: str) -> list[str]:
123        """
124        Run an MCRCON command and return the output
125
126        This method runs a command against the configured MCRCON target,
127        strips out all of the ANSI characters, and any irrelevant lines of
128        data, and returns the output as a list of strings.
129
130        Args:
131            cmd: The command you want to run
132
133        Returns:
134            A list of strings where each string in the list is a line of
135            output from the command
136
137        Raises:
138            ValueError when the command fails to run
139        """
140        if not self.password:
141            raise ValueError('MCRCON_PASS is unset!')
142
143        command: list[str] = [
144            '/usr/local/bin/mcrcon', '-p', self.password, cmd
145        ]
146        try:
147            output: subprocess.CompletedProcess = subprocess.run(
148                command, capture_output=True, check=True
149            )
150            if output.returncode > 0:
151                raise ValueError(f'[WARN] {output.stderr.decode()}')
152
153            data: str = self._ansi_decode(output.stdout.decode())
154            data = self._mcrcon_strip(data)
155
156            return data.splitlines()
157
158        except subprocess.CalledProcessError as err:
159            stdout: list[str] = err.stdout.decode().splitlines()
160            stderr: list[str] = err.stderr.decode().splitlines()
161
162            msg: list[str] = ['[ERROR]', ''.join(stdout)]
163            if len(stderr) > 0:
164                msg.append('::')
165                msg.append(''.join(stderr))
166
167            raise ValueError(' '.join(msg)) from err
168

171        Get the count of logged in players

173        This method returns the total number of players online as found

198    def get_online_players(
199            self, players: list[dict[str, str]]) -> list[dict[str, Any]]:
200        """
201        Get the names of players who are logged in
202
203        This method returns which players are online and offline as found
204        from the `list` MCRCON command.
205
206        Args:
207            players: A list of dictionaries containing the player names
208            from the whitelist
209
210        Returns:
211            A list of dictionaries where each dictionary is a player and
212            their online status, like this:
213            {
214                'player': playername1234,
215                'count': 0 # 0 if offline, 1 if online
216            }
217
218        Raises:
219            None
220        """
221        logged_in_players: list[dict[str, Any]] = []
222
223        for player in players:
224            for key, value in player.items():
225                if key == 'name':
226                    logged_in: dict[str, Any] = {
227                        'player': value,
228                        'count': 0
229                    }
230                    logged_in_players.append(logged_in)
231
232        pattern: str = r'^.*?online\:\s+([A-z0-9\,\s]+)'
233        try:
234            output: list[str] = self._run_command('list')
235            matches = re.match(pattern, output[0])
236            if not matches:
237                return logged_in_players
238
239            for player in matches.group(1).split(','):
240                name: str = player.strip()
241                for logged_in in logged_in_players:
242                    if logged_in['player'] == name:
243                        logged_in['count'] = 1
244
245        except ValueError as err:
246            print(err)
247
248        return logged_in_players
